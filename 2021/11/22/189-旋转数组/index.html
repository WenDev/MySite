func rotate(nums []int, k int) {
    k %= len(nums)
    reverse(nums)
    reverse(nums[:k])
    reverse(nums[k:])
}

func reverse(arr []int) {
    i, j := 0, len(arr)-1
    for i < j {
        temp := arr[i]
        arr[i] = arr[j]
        arr[j] = temp
        i++
        j--
    }
}
